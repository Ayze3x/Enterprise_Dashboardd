Enterprise Authentication Flow



RBAC Middleware
import { Request, Response, NextFunction } from "express" import jwt from "jsonwebtoken"

interface JwtPayload { id: string
role: "Admin" | "Manager" | "Analyst" | "Viewer"
}

export interface AuthRequest extends Request { user?: JwtPayload
}

export const authenticate = ( req: AuthRequest,
res: Response, next: NextFunction
) => {
try {
const token = req.headers.authorization?.split(" ")[1]

if (!token) {
return res.status(401).json({ success: false,
error: "Unauthorized"
})
}

const decoded = jwt.verify( token,
process.env.JWT_SECRET as string
) as JwtPayload req.user = decoded
next()
} catch (error) {
return res.status(401).json({ success: false,
error: "Invalid token"
})
}
}
